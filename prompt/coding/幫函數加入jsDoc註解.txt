const useFindPageSequenceIndex = (docmFileSeq: string, docmNo: string, docmPageSeq: string): number => {
  const { scanFileList } = useUploadPhotos()
  const key = `${docmNo}-${docmFileSeq}-${docmPageSeq}`
  const index = useMemo(() => {
    return _.findIndex(scanFileList, (e) => `${e.docmNo}-${e.docmFileSeq}-${e.docmPageSeq}` === key)
  }, [scanFileList, key])
  return index
}
